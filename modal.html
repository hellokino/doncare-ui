对话框的炼成

请按这个格式制作对话框

<div id="modal-alert" class="modal">
    <!-- 对话框的标题 -->
    <div class="modal-header">
        对话框 modal
    </div>
    <div class="modal-main">
        <!-- 对话框的主体内容 -->
        这是对话框<br />
        Tada! I'm a modal.

        <!-- 对话框的操作栏 -->
        <div class="modal-option">
            <a href="javascript:;" onclick="DC.hideModal('#modal-alert')" class="btn cross btn-primary">
                <span><i class="icon-ok"></i>确定</span>
            </a>
        </div>

        <!-- 这是对话框背景，请保留 -->
        <span class="modal-rolling-bg"><i class="icon-cog"></i></span>
        <span class="modal-rocking-bg"><i class="icon-reload"></i></span>
        <span class="modal-rolling-bg2"><i class="icon-cog"></i></span>
    </div>
</div>

<!-- 触发一个对话框 -->
<a href="javascript:;" onclick="DC.showModal('#modal-alert')" class="btn">
    <span><i class="icon-dialog"></i>点我开启对话框</span>
</a>
